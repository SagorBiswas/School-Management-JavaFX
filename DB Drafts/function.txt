CREATE OR REPLACE FUNCTION SHOW_GRADE(EXM_ID IN VARCHAR2,STD_ID IN VARCHAR2)
RETURN VARCHAR2 IS
COUNT NUMBER;
TOTAL NUMBER ;
AVG NUMBER ;
NUM_SUBJECT NUMBER ;
GRADE VARCHAR2(100) ;

BEGIN
SELECT SUM(MARKS) INTO TOTAL
FROM RESULT
WHERE STUDENT_ID = STD_ID AND EXAM_ID=EXM_ID;

SELECT COUNT(MARKS) INTO NUM_SUBJECT 
FROM RESULT
WHERE STUDENT_ID = STD_ID AND EXAM_ID=EXM_ID;

AVG := (TOTAL / NUM_SUBJECT) ;
IF AVG >= 80 THEN
GRADE := 'A+' ;

ELSIF AVG >=70  THEN
GRADE := 'A';

ELSIF AVG >=65  THEN
GRADE := 'A-';

ELSIF AVG >=60  THEN
GRADE := 'B';

ELSIF AVG >=55  THEN
GRADE := 'C';

ELSIF AVG >=50  THEN
GRADE := 'D';

ELSIF AVG >=40  THEN
GRADE := 'E';
ELSE GRADE:='F';
END IF ;
RETURN GRADE ; --return the message

END ;
